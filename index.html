Password rule:


const PassWord = new Rule({
    type: 'password',
    minLength: 8,
    maxLength: 10,
    matchesOneOf: ['@', '_', '-', '.', '?', '$'],
    numbers: 1,
    uppercase: 1,
  });

Url rule:


const url = new Rule({
  type: 'url',
  domain: domain => domain === 'github.com',
  protocol: protocol => protocol === 'https',
});
